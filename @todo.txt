- per gestire il nome nell header da mobile 
'use client';
import { usePathname } from 'next/navigation';
import { menuItems } from '../data/menuItems'; // il tuo array condiviso

export default function MobileHeader() {
  const pathname = usePathname();

  // cerca il titolo corrispondente alla route
  const currentItem = menuItems.find(item => item.link === pathname);
  const title = currentItem?.title || 'Dashboard';

  return (
    <header className="mobile-header flex items-center justify-between p-4 bg-white shadow-md md:hidden">
      <button className="back-button">← Torna indietro</button>
      <h1 className="text-lg font-semibold">{title}</h1>
      <button className="profile-button">Profilo</button>
    </header>
  );
}
